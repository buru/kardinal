def destroy

def update
  @user = current_user
  @contact_detail = @user.contact_details.first

  if @user.update_attributes(params[:user]) &&
    @contact_detail.update_attributes(params[:contact_details])

    flash[:notices] = ["Your profile was successfully updated!"]
    redirect_to me_url
  else
    render :edit
  end
end
